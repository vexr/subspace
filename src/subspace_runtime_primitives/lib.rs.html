130pub mod time {
131    /// Expected block time in milliseconds.
132    ///
133    /// Since Subspace is probabilistic this is the average expected block time that
134    /// we are targeting. Blocks will be produced at a minimum duration defined
135    /// by `SLOT_DURATION`, but some slots will not be allocated to any
136    /// farmer and hence no block will be produced. We expect to have this
137    /// block time on average following the defined slot duration and the value
138    /// of `c` configured for Subspace (where `1 - c` represents the probability of
139    /// a slot being empty).
140    /// This value is only used indirectly to define the unit constants below
141    /// that are expressed in blocks. The rest of the code should use
142    /// `SLOT_DURATION` instead (like the Timestamp pallet for calculating the
143    /// minimum period).
144    ///
145    /// Based on:
146    /// <https://research.web3.foundation/en/latest/polkadot/block-production/Babe.html#-6.-practical-results>
147    pub const MILLISECS_PER_BLOCK: u64 = 6000;
148    /// Approximate number of block in a minute.
149    pub const BLOCKS_IN_AN_MINUTE: u32 = (60 * 1000) / MILLISECS_PER_BLOCK as u32;
150    /// Approximate number of blocks in an hour.
151    pub const BLOCKS_IN_AN_HOUR: u32 = 60 * BLOCKS_IN_AN_MINUTE;
152    /// Approximate number of blocks in a day.
153    pub const BLOCKS_IN_A_DAY: u32 = 24 * BLOCKS_IN_AN_HOUR;
154}

234parameter_types! {
235    /// The portion of the `NORMAL_DISPATCH_RATIO` that we adjust the fees with. Blocks filled less
236    /// than this will decrease the weight and more will increase.
237    pub const TargetBlockFullness: Perquintill = Perquintill::from_percent(50);
238    /// The adjustment variable of the runtime. Higher values will cause `TargetBlockFullness` to
239    /// change the fees more rapidly.
240    pub AdjustmentVariable: Multiplier = Multiplier::saturating_from_rational(75, 1_000_000);
241    /// Minimum amount of the multiplier. This value cannot be too low. A test case should ensure
242    /// that combined with `AdjustmentVariable`, we can recover from the minimum.
243    /// See `multiplier_can_grow_from_zero`.
244    pub MinimumMultiplier: Multiplier = Multiplier::saturating_from_rational(1, 10u128);
245    /// The maximum amount of the multiplier.
246    pub MaximumMultiplier: Multiplier = Bounded::max_value();
247}

285/// Adjusted XDM Weight to fee Conversion.
286pub struct XdmAdjustedWeightToFee<T>(PhantomData<T>);
287impl<T: pallet_transaction_payment::Config> WeightToFee for XdmAdjustedWeightToFee<T> {
288    type Balance = OnChargeTransactionBalance<T>;
289
290    fn weight_to_fee(weight: &Weight) -> Self::Balance {
291        // the adjustable part of the fee.
292        let unadjusted_weight_fee = pallet_transaction_payment::Pallet::<T>::weight_to_fee(*weight);
293        let multiplier = NextFeeMultiplier::<T>::get();
294        // final adjusted weight fee.
295        multiplier.saturating_mul_int(unadjusted_weight_fee)
296    }
297}

316#[cfg(feature = "testing")]
317pub mod tests_utils {
318    use frame_support::dispatch::DispatchClass;
319    use frame_support::weights::Weight;
320    use frame_system::limits::BlockWeights;
321    use pallet_transaction_payment::{Multiplier, MultiplierUpdate};
322    use sp_runtime::BuildStorage;
323    use sp_runtime::traits::{Convert, Get};
324    use std::marker::PhantomData;
325
326    pub struct FeeMultiplierUtils<Runtime, BlockWeightsGetter>(
327        PhantomData<(Runtime, BlockWeightsGetter)>,
328    );
329
330    impl<Runtime, BlockWeightsGetter> FeeMultiplierUtils<Runtime, BlockWeightsGetter>
331    where
332        Runtime: frame_system::Config + pallet_transaction_payment::Config,
333        BlockWeightsGetter: Get<BlockWeights>,
334    {
335        fn max_normal() -> Weight {
336            let block_weights = BlockWeightsGetter::get();
337            block_weights
338                .get(DispatchClass::Normal)
339                .max_total
340                .unwrap_or(block_weights.max_block)
341        }
342
343        fn min_multiplier() -> Multiplier {
344            <Runtime as pallet_transaction_payment::Config>::FeeMultiplierUpdate::min()
345        }
346
347        fn target() -> Weight {
348            <Runtime as pallet_transaction_payment::Config>::FeeMultiplierUpdate::target()
349                * Self::max_normal()
350        }
351
352        // update based on runtime impl.
353        fn runtime_multiplier_update(fm: Multiplier) -> Multiplier {
354            <Runtime as pallet_transaction_payment::Config>::FeeMultiplierUpdate::convert(fm)
355        }
356
357        fn run_with_system_weight<F>(w: Weight, assertions: F)
358        where
359            F: Fn(),
360        {
361            let mut t: sp_io::TestExternalities = frame_system::GenesisConfig::<Runtime>::default()
362                .build_storage()
363                .unwrap()
364                .into();
365            t.execute_with(|| {
366                frame_system::Pallet::<Runtime>::set_block_consumed_resources(w, 0);
367                assertions()
368            });
369        }
370
371        // The following function is taken from test with same name from
372        // https://github.com/paritytech/polkadot-sdk/blob/91851951856b8effe627fb1d151fe336a51eef2d/substrate/bin/node/runtime/src/impls.rs#L234
373        // with some small surface changes.
374        pub fn multiplier_can_grow_from_zero()
375        where
376            Runtime: pallet_transaction_payment::Config,
377            BlockWeightsGetter: Get<BlockWeights>,
378        {
379            // if the min is too small, then this will not change, and we are doomed forever.
380            // the block ref time is 1/100th bigger than target.
381            Self::run_with_system_weight(
382                Self::target().set_ref_time((Self::target().ref_time() / 100) * 101),
383                || {
384                    let next = Self::runtime_multiplier_update(Self::min_multiplier());
385                    assert!(
386                        next > Self::min_multiplier(),
387                        "{:?} !> {:?}",
388                        next,
389                        Self::min_multiplier()
390                    );
391                },
392            );
393
394            // the block proof size is 1/100th bigger than target.
395            Self::run_with_system_weight(
396                Self::target().set_proof_size((Self::target().proof_size() / 100) * 101),
397                || {
398                    let next = Self::runtime_multiplier_update(Self::min_multiplier());
399                    assert!(
400                        next > Self::min_multiplier(),
401                        "{:?} !> {:?}",
402                        next,
403                        Self::min_multiplier()
404                    );
405                },
406            )
407        }
408    }
